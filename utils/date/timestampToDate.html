--- // 时间格式化 formatMillisecond : function() { var els = $(".dateTd"); var dt = new Date(); if (els) { for (var i = 0; i < els.length; i++) { var eText = els[i].innerHTML; if (eText == '' || eText == '0' || isNaN(eText)) { els[i].innerHTML = '--'; continue; } dt.setTime(eText); els[i].innerHTML = dt.format('yyyy-MM-dd hh:mm'); } } }
